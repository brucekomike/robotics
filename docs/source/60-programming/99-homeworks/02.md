from sympy import symbols, sqrt, acos, cos, sin, pi

# Define symbols
x, y, z = symbols('x y z')  # End effector position
L, l, Rb, Re = symbols('L l Rb Re')  # Robot parameters
theta1, theta2, theta3 = symbols('theta1 theta2 theta3')  # Joint angles

# Calculate the coordinates of the arm bases
x1 = x - Re * cos(theta1)
y1 = y - Re * sin(theta1)
z1 = z

x2 = x - Re * cos(theta2)
y2 = y - Re * sin(theta2)
z2 = z

x3 = x - Re * cos(theta3)
y3 = y - Re * sin(theta3)
z3 = z

# Calculate distances from base to end effector
d1 = sqrt(x1**2 + y1**2 + z1**2)
d2 = sqrt(x2**2 + y2**2 + z2**2)
d3 = sqrt(x3**2 + y3**2 + z3**2)

# Calculate joint angles using inverse kinematics
theta1_sol = acos((L**2 + d1**2 - l**2) / (2 * L * d1))
theta2_sol = acos((L**2 + d2**2 - l**2) / (2 * L * d2))
theta3_sol = acos((L**2 + d3**2 - l**2) / (2 * L * d3))

# Display the solutions
theta1_sol, theta2_sol, theta3_sol
